  iMidiStart = p4
  iMidiEnd = p5
  kDb = linseg:k(-10,p3/2,-20)
  kMidi = linseg:k(iMidiStart,p3/3,iMidiEnd)
  aSine = poscil:a(ampdb(kDb),mtof(kMidi))
  aOut = linen:a(aSine,0,p3,1)
  outall(aOut)
  
  if (p6 > 0) then
	schedule("Hello",p3,p3+1,p4-1,p5+2,p6-1)
    print(p6)
  endif
endin
schedule("Hello", 0, 2, 72, 68, 9)
